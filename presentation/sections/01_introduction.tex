% Why This Paper?
\begin{frame}{Why This Paper?}
    \begin{itemize}
        \item Published in \textit{Nature}, the most prestigious scientific journal
        \item \textbf{\textasciitilde70,000 citations}: one of the most cited CS papers ever
        \item Written by the three researchers most responsible for the field
        \item All three: \textbf{2018 Turing Award} (Nobel Prize of CS)
        \item Hinton: \textbf{2024 Nobel Prize in Physics}
    \end{itemize}
    
    \vspace{1em}
    The paper is simultaneously:
    \begin{itemize}
        \item A technical survey
        \item A historical account written by its participants
        \item An argument to a skeptical research community
    \end{itemize}
\end{frame}
